017package net.ljcomputing;
018
019/**
020 * Implementation of a utility class to create SQL statements 
021 * that can be used with JDBC implementation.
022 * 
023 * @author James G. Willmore
024 *
025 */
026public final class SqlUtils {
027
028  /** The Constant AT_LEAST_ONE - suggested by PMD plugin. */
029  private final static int AT_LEAST_ONE = 1;
030  
031  /** The Constant COMMA - suggested by PMD plugin. */
032  private final static char COMMA = ',';
033  
034  /** The Constant QUESTION_MARK - suggested by PMD plugin. */
035  private final static char QUESTION_MARK = '?';
036  
037  /** The Constant EQUALS - suggested by PMD plugin. */
038  private final static char EQUALS = '=';
039  
040  /** The Constant WHERE - suggested by PMD plugin. */
041  private final static String WHERE = " where ";
042  
043  /** The Constant SET - suggested by PMD plugin. */
044  private final static String SET = " set ";
045  
046  /** The Constant VALUES_TOKEN - suggested by PMD plugin. */
047  private final static String VALUES_TOKEN = " values(";
048  
049  /**
050   * Instantiates a new sql utils.
051   */
052  private SqlUtils() {}
053
054  /**
055   * Builds the columns values used in various SQL statements based upon the given columns.
056   *
057   * @param columns the columns
058   * @return the string
059   */
060  private static String buildColumns(final String... columns) {
061    final StringBuffer buf = new StringBuffer();
062
063    for (final String column : columns) {
064      buf.append(column).append(COMMA);
065    }
066
067    if (buf.length() > AT_LEAST_ONE) {
068      buf.reverse().deleteCharAt(0).reverse();
069    }
070
071    return buf.toString();
072  }
073
074  /**
075   * Builds the place holders used in a PreparedStatement based upon the given columns.
076   *
077   * @param columns the columns
078   * @return the string
079   */
080  private static String buildPlaceholders(final String... columns) {
081    final StringBuffer buf = new StringBuffer();
082
083    for (int c = 0; c < columns.length; c++) {
084      buf.append(QUESTION_MARK).append(COMMA);
085    }
086
087    if (buf.length() > AT_LEAST_ONE) {
088      buf.reverse().deleteCharAt(0).reverse();
089    }
090
091    return buf.toString();
092  }
093
094  /**
095   * Builds the SET column values; part of an UPDATE SQL statement.
096   *
097   * @param columns the columns
098   * @return the string
099   */
100  private static String buildSetColumnValues(final String... columns) {
101    final StringBuffer buf = new StringBuffer();
102
103    for (final String column : columns) {
104      buf.append(column).append(EQUALS).append(QUESTION_MARK).append(COMMA);
105    }
106
107    if (buf.length() > AT_LEAST_ONE) {
108      buf.reverse().deleteCharAt(0).reverse();
109    }
110
111    return buf.toString();
112  }
113
114  /**
115   * Builds the INSERT SQL statement used in a PreparedStatement.
116   *
117   * @param table the table
118   * @param columns the columns
119   * @return the string
120   */
121  public static String buildInsertStatement(final String table,
122      final String... columns) {
123    final StringBuffer buf = new StringBuffer("insert into ");
124
125    buf.append(table).append('(').append(buildColumns(columns)).append(')')
126        .append(VALUES_TOKEN).append(buildPlaceholders(columns)).append(')');
127
128    return buf.toString();
129  }
130
131  /**
132   * Builds the UPDATE SQL statement used in a PreparedStatement.
133   *
134   * @param table the table
135   * @param primaryKeyColumn the primary key column
136   * @param columns the columns
137   * @return the string
138   */
139  public static String buildUpdateStatement(final String table,
140      final String primaryKeyColumn, final String... columns) {
141    final StringBuffer buf = new StringBuffer("update ");
142
143    buf.append(table).append(SET).append(buildSetColumnValues(columns))
144        .append(WHERE).append(buildSetColumnValues(primaryKeyColumn));
145
146    return buf.toString();
147  }
148
149  /**
150   * Builds the DELETE SQL statement used in a PreparedStatement.
151   *
152   * @param table the table
153   * @param primaryKeyColumn the primary key column
154   * @return the string
155   */
156  public static String buildDeleteStatement(final String table,
157      final String primaryKeyColumn) {
158    final StringBuffer buf = new StringBuffer("delete from ");
159
160    buf.append(table).append(WHERE)
161        .append(buildSetColumnValues(primaryKeyColumn));
162
163    return buf.toString();
164  }
165
166}
