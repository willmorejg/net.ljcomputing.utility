017package net.ljcomputing;
018
019import java.util.Locale;
020
021/**
022 * A String utility class.
023 * 
024 * @author James G. Willmore
025 * 
026 */
027public final class StringUtils {
028
029  /** The recognized separators. */
030  private final static char[] SEPARATORS = new char[] { '_', '.' };
031
032  /**
033   * Instantiates a new string utils.
034   */
035  private StringUtils() {
036  }
037
038  /**
039   * <p>Return the camel case representation the incoming String. </p> <p>NOTE:
040   * this method will return a camel case representation of the String suitable
041   * for various concatenation algorithms. </p> <p>For example, "Yearly Income"
042   * becomes "YearlyIncome", which in turn could be used to create another
043   * String "getYearlyIncome". </p>
044   *
045   * @param value the value @return the string
046   * @return the string
047   */
048  public static final String toCamelCase(final String value) {
049    return camelCase(value.toLowerCase(Locale.ENGLISH));
050  }
051
052  /**
053   * <p>Return the camel case representation the incoming String. </p> <p>NOTE:
054   * this method will return a camel case representation of the String suitable
055   * for various concatenation algorithms. </p> <p>For example, "Yearly Income"
056   * becomes "YearlyIncome", which in turn could be used to create another
057   * String "getYearlyIncome". </p>
058   *
059   * @param value the value
060   * @return the string
061   */
062  private static String camelCase(final String value) {
063    final StringBuffer buf = new StringBuffer();
064
065    if (value != null && !value.trim().isEmpty()) {
066      final char[] chars = value.trim().toCharArray();
067
068      buf.append(String.valueOf(chars[0]).toUpperCase(Locale.ENGLISH));
069
070      for (int i = 1; i < chars.length;) {
071        final boolean foundSeparator = isSeparator(chars[i]);
072
073        if (!foundSeparator) {
074          buf.append(chars[i]);
075        }
076
077        if (foundSeparator && i < chars.length) {
078          buf.append(String.valueOf(chars[++i]).toUpperCase(Locale.ENGLISH));
079        }
080
081        i++;
082      }
083    }
084
085    return buf.toString();
086  }
087
088  /**
089   * Checks if is separator.
090   *
091   * @param character the character
092   * @return true, if is separator
093   */
094  private static boolean isSeparator(final char character) {
095    boolean result = false;
096
097    for (final char check : SEPARATORS) {
098      if (check == character) {
099        result = true;
100        break;
101      }
102    }
103
104    return result;
105  }
106
107  /**
108   * <p>Return the member case representation the incoming String. </p> <p>NOTE:
109   * this method will return a member case representation of the String suitable
110   * for various uses of java.reflect.*. </p> <p>For example, "Yearly Income"
111   * becomes "yearlyIncome", which in turn could be used to access a Class'
112   * member field called by the same name. </p>
113   *
114   * @param value the value
115   * @return the string
116   */
117  public static final String toMemberCase(final String value) {
118    return memberCase(value.toLowerCase(Locale.ENGLISH));
119  }
120
121  /**
122   * <p>Return the member case representation the incoming String. </p> <p>NOTE:
123   * this method will return a member case representation of the String suitable
124   * for various uses of java.reflect.*. </p> <p>For example, "Yearly Income"
125   * becomes "yearlyIncome", which in turn could be used to access a Class'
126   * member field called by the same name. </p>
127   *
128   * @param value the value
129   * @return the string
130   */
131  private static String memberCase(final String value) {
132    final StringBuffer buf = new StringBuffer();
133
134    buf.append(camelCase(value));
135    buf.replace(0, 1, buf.substring(0, 1).toLowerCase(Locale.ENGLISH));
136
137    return buf.toString();
138  }
139  
140  /**
141   * Checks if the given String value is null, or is equal to an empty String <i>when trimmed</i>.
142   *
143   * @param value the value
144   * @return true, if is null, or empty when trimmed
145   */
146  public static final boolean isNullOrEmpty(final String value) {
147    return value == null || "".equals(value.trim());
148  }
149}
